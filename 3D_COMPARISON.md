export const config = {
  runtime: 'nodejs',  // Serverless
  maxDuration: 300    // 5 –º–∏–Ω—É—Ç

export default async function handler(req, res) {
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }

  try {
    const { image_url } = req.body;
    
    if (!image_url) {
      return res.status(400).json({ error: 'image_url required' });
    }

    // Step 1: Submit 3D generation task
    const submitResponse = await fetch('https://fal.run/fal-ai/tripo3d/image-to-3d', {
      method: 'POST',
      headers: {
        'Authorization': `Key ${process.env.FAL_API_KEY}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        image_url: image_url,
        model_format: 'glb',
        face_limit: 10000,
        texture_resolution: 1024
      })
    });

    const submitData = await submitResponse.json();
    const requestId = submitData.request_id;

    // Step 2: Poll for completion (max 3 minutes)
    let status = 'IN_QUEUE';
    let attempts = 0;
    const maxAttempts = 90; // 3 minutes (2 sec intervals)

    while (status !== 'COMPLETED' && attempts < maxAttempts) {
      await new Promise(resolve => setTimeout(resolve, 2000)); // Wait 2 seconds

      const statusResponse = await fetch(
        `https://fal.run/fal-ai/tripo3d/image-to-3d/requests/${requestId}/status`,
        {
          headers: {
            'Authorization': `Key ${process.env.FAL_API_KEY}`
          }
        }
      );

      const statusData = await statusResponse.json();
      status = statusData.status;

      if (status === 'COMPLETED') {
        console.log('[3D SINGLE] Generation completed');
        return res.status(200).json({
          success: true,
          model_url: statusData.model_url,
          preview_url: statusData.preview_url,
          face_count: statusData.face_count,
          cost: 0.33,
          processing_time: attempts * 2
        });
      }

      if (status === 'FAILED') {
        console.error('[3D SINGLE] Generation failed:', statusData.error);
        return res.status(500).json({
          error: '3D generation failed',
          details: statusData.error
        });
      }

      attempts++;
    }

    // Timeout after 3 minutes
    return res.status(408).json({
      error: 'Generation timeout',
      message: '3D generation took too long. Please try again.'
    });

  } catch (error) {
    console.error('[3D SINGLE] Error:', error);
    return res.status(500).json({
      error: 'Internal server error',
      details: error.message
    });
  }
}
